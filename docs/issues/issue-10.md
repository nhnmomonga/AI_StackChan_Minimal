```cpp
// メモリリークテスト
void test_memory_leak() {
  for (int i = 0; i < 100; i++) {
    start_talking();
    delay(30000); // 30秒待機
    size_t free_heap = esp_get_free_heap_size();
    Serial.printf("Iteration %d: Free heap: %d\n", i, free_heap);
  }
}

// レスポンス時間テスト
void test_response_time() {
  unsigned long start = millis();
  start_talking();
  unsigned long end = millis();
  Serial.printf("Total response time: %lu ms\n", end - start);
}
```
